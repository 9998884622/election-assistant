function sendMessage() { const input = document.getElementById('userInput'); const originalText = input.value.trim(); const text = originalText.toLowerCase(); if (!text) return; appendMessage(originalText, true); input.value = ''; setTimeout(() => { // PRIORITY 1: Check for Registration first if (text.includes('reg')) { handleBtn('Register'); } // PRIORITY 2: Check for Deadlines/Dates else if (text.includes('date') || text.includes('when') || text.includes('time')) { handleBtn('Dates'); } // PRIORITY 3: Check for ID else if (text.includes('id') || text.includes('license') || text.includes('document')) { handleBtn('ID'); } // PRIORITY 4: General Voting Process else if (text.includes('vote') || text.includes('how') || text.includes('process')) { handleBtn('Process'); } else { appendMessage("I'm here to help! Try asking about 'registration', 'deadlines', or 'voting process'.", false); } }, 600); }
